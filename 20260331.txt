    <!-- 應用程式邏輯 (Vanilla JS 單頁應用) -->
    <script>
        /** * 設定區：請將此處替換為您部署 Google Apps Script 後獲得的 Web App URL
         * @type {string}
         */
        const GAS_API_URL = "https://script.google.com/macros/s/AKfycbw8pc2DUZNLCNA4jBVPsCBvFtBmikvb_FWfFPygl_OUbfS4EXPoZhR7hW2PpNb1KlQSyA/exec"; 

        // === 內部狀態管理 ===
        const app = {
            data: {
                products: [],
                settings: {
                    site_name: "豪豪出包",
                    brand_description: "Sourdough Cracker \n酸種酵母餅乾",
                    order_instructions: "載入中...",
                    brand_statement: "載入中...",
                    shipping_fee: 60,
                    bank_account: "",
                    bank_reminder: ""
                },
                cart: {}, // { productId: quantity }
                currentPage: 'loading' // 'loading', 'home', 'checkout', 'success'
            },

            // 系統初始化
            async init() {
                try {
                    // 如果沒有設定 API URL，使用模擬測試資料以供開發預覽
                    if (!GAS_API_URL) {
                        this.showToast("目前使用測試資料（請記得設定 GAS URL）");
                        this.loadMockData();
                    } else {
                        const response = await fetch(GAS_API_URL + "?action=getData");
                        const result = await response.json();
                        if(result.status === 'success') {
                            this.data.products = result.data.products;
                            this.data.settings = result.data.settings;
                        } else {
                            throw new Error("API 回傳錯誤");
                        }
                    }
                    this.renderSettings();
                    this.renderProducts();
                    this.navigate('home');
                } catch (error) {
                    console.error("初始化失敗:", error);
                    this.showToast("資料載入失敗，請稍後再試或聯繫客服。");
                    // 發生錯誤也載入模擬資料以免畫面空白
                    this.loadMockData();
                    this.renderSettings();
                    this.renderProducts();
                    this.navigate('home');
                }
            },

            // 路由切換
            navigate(page) {
                this.data.currentPage = page;
                window.scrollTo(0, 0);
                
                document.getElementById('loadingView').classList.add('hidden');
                document.getElementById('homeView').classList.add('hidden');
                document.getElementById('checkoutView').classList.add('hidden');
                document.getElementById('successView').classList.add('hidden');
                document.getElementById('floatingCartBtn').style.display = 'none';

                if (page === 'home') {
                    document.getElementById('homeView').classList.remove('hidden');
                    this.updateCartUI();
                } else if (page === 'checkout') {
                    if (Object.keys(this.data.cart).length === 0) {
                        this.showToast("購物車是空的喔！");
                        this.navigate('home');
                        return;
                    }
                    document.getElementById('checkoutView').classList.remove('hidden');
                    this.renderCheckoutCart();
                } else if (page === 'success') {
                    document.getElementById('successView').classList.remove('hidden');
                }
            },

            // 渲染全域設定文字
            renderSettings() {
                document.title = `${this.data.settings.site_name} | 酵母餅乾訂購`;
                document.getElementById('brandDescTitle').innerText = this.data.settings.site_name;
                document.getElementById('orderInstructions').innerText = this.data.settings.order_instructions;
                document.getElementById('brandStatement').innerText = this.data.settings.brand_statement;
                document.getElementById('bankAccount').innerText = this.data.settings.bank_account;
                document.getElementById('bankReminder').innerText = this.data.settings.bank_reminder;
            },

            // 渲染商品列表
            renderProducts() {
                const container = document.getElementById('productsContainer');
                container.innerHTML = '';
                
                const visibleProducts = this.data.products
                    .filter(p => String(p.is_visible).toUpperCase() === 'TRUE')
                    .sort((a, b) => parseInt(a.sort_order) - parseInt(b.sort_order));

                visibleProducts.forEach(product => {
                    // Initialize cart state
                    if(this.data.cart[product.id] === undefined) this.data.cart[product.id] = 0;
                    
                    const priceNum = parseInt(product.price.replace(/\D/g, '')) || 0; // fallback parse
                    const imageSrc = product.main_image_url || 'https://images.unsplash.com/photo-1600857544200-b2f666a9a2ec?ixlib=rb-4.0.3&auto=format&fit=crop&w=500&q=80';

                    const html = `
                        <div class="flex flex-col md:flex-row gap-8 items-start">
                            <!-- 圖片區塊 -->
                            <div class="w-full md:w-1/2 aspect-square rounded-2xl overflow-hidden bg-bglight relative">
                                <img src="${imageSrc}" alt="${product.name}" class="w-full h-full object-cover" onerror="this.src='https://images.unsplash.com/photo-1600857544200-b2f666a9a2ec?ixlib=rb-4.0.3&auto=format&fit=crop&w=500&q=80'">
                                <div class="absolute bottom-4 right-4 bg-white/90 backdrop-blur text-primary text-xs font-bold px-3 py-1 rounded-full shadow-sm">
                                    ${product.weight_grams}g
                                </div>
                            </div>
                            
                            <!-- 資訊與購買區塊 -->
                            <div class="w-full md:w-1/2 flex flex-col justify-center">
                                <h2 class="text-2xl font-bold mb-2 text-primary">${product.name}</h2>
                                <p class="text-lg font-medium text-accent1 mb-6">${product.price}</p>
                                
                                <div class="text-secondary text-sm leading-relaxed mb-6 whitespace-pre-line">
                                    ${product.description}
                                </div>
                                
                                <div class="bg-bglight/50 p-4 rounded-xl mb-8">
                                    <h4 class="text-xs font-bold text-graytext mb-2">成分說明</h4>
                                    <p class="text-xs text-secondary whitespace-pre-line">${product.ingredients}</p>
                                </div>
                                
                                <!-- 數量選擇器 -->
                                <div class="flex items-center gap-4 border border-gray-200 rounded-xl p-2 w-max mb-2">
                                    <button onclick="app.updateQuantity(${product.id}, -1, ${product.max_quantity})" class="w-10 h-10 flex items-center justify-center text-graytext hover:text-primary transition-colors disabled:opacity-30" id="btn-minus-${product.id}">
                                        <i class="fa-solid fa-minus"></i>
                                    </button>
                                    <span class="w-10 text-center font-bold text-lg" id="qty-${product.id}">0</span>
                                    <button onclick="app.updateQuantity(${product.id}, 1, ${product.max_quantity})" class="w-10 h-10 flex items-center justify-center text-graytext hover:text-primary transition-colors disabled:opacity-30" id="btn-plus-${product.id}">
                                        <i class="fa-solid fa-plus"></i>
                                    </button>
                                </div>
                                <p class="text-xs text-graytext mt-1">每人限購 ${product.max_quantity} 包</p>
                            </div>
                        </div>
                    `;
                    container.insertAdjacentHTML('beforeend', html);
                });
                this.updateCartUI();
            },

            // 更新商品數量
            updateQuantity(productId, change, maxQty) {
                let current = this.data.cart[productId] || 0;
                let next = current + change;
                
                if (next < 0) next = 0;
                if (next > maxQty) {
                    this.showToast(`此款餅乾每人限購 ${maxQty} 包喔！`);
                    next = maxQty;
                }
                
                this.data.cart[productId] = next;
                document.getElementById(`qty-${productId}`).innerText = next;
                
                this.updateCartUI();
            },

            // 更新全域購物車狀態與 UI (包含小圖示與懸浮按鈕)
            updateCartUI() {
                let totalItems = 0;
                let totalPrice = 0;

                this.data.products.forEach(p => {
                    const qty = this.data.cart[p.id] || 0;
                    totalItems += qty;
                    if(qty > 0) {
                        const price = parseInt(p.price.replace(/\D/g, '')) || 0;
                        totalPrice += price * qty;
                    }
                    
                    // Update buttons state
                    const btnMinus = document.getElementById(`btn-minus-${p.id}`);
                    const btnPlus = document.getElementById(`btn-plus-${p.id}`);
                    if(btnMinus) btnMinus.disabled = qty <= 0;
                    if(btnPlus) btnPlus.disabled = qty >= p.max_quantity;
                });

                // Header Badge
                const badge = document.getElementById('cartBadge');
                if (totalItems > 0) {
                    badge.innerText = totalItems;
                    badge.classList.remove('hidden');
                } else {
                    badge.classList.add('hidden');
                }

                // Floating Button
                const floatingBtn = document.getElementById('floatingCartBtn');
                if (this.data.currentPage === 'home' && totalItems > 0) {
                    document.getElementById('floatingCartTotal').innerText = `NT$ ${totalPrice}`;
                    floatingBtn.style.display = 'flex';
                } else {
                    floatingBtn.style.display = 'none';
                }
            },

            // 結帳頁面的購物車渲染
            renderCheckoutCart() {
                const container = document.getElementById('cartItemsContainer');
                container.innerHTML = '';
                
                let subtotal = 0;

                this.data.products.forEach(p => {
                    const qty = this.data.cart[p.id] || 0;
                    if (qty > 0) {
                        const price = parseInt(p.price.replace(/\D/g, '')) || 0;
                        const itemTotal = price * qty;
                        subtotal += itemTotal;
                        
                        const html = `
                            <div class="flex justify-between items-center py-3 border-b border-gray-200/50 last:border-0">
                                <div>
                                    <p class="font-medium text-primary">${p.name}</p>
                                    <p class="text-xs text-graytext">${p.price} x ${qty}</p>
                                </div>
                                <div class="font-bold text-secondary">
                                    NT$ ${itemTotal}
                                </div>
                            </div>
                        `;
                        container.insertAdjacentHTML('beforeend', html);
                    }
                });

                const shipping = parseInt(this.data.settings.shipping_fee) || 60;
                const total = subtotal + shipping;

                document.getElementById('cartSubtotal').innerText = `NT$ ${subtotal}`;
                document.getElementById('cartShipping').innerText = `NT$ ${shipping}`;
                document.getElementById('cartTotal').innerText = `NT$ ${total}`;
                document.getElementById('successTotal').innerText = `NT$ ${total}`;
            },

            // 送出訂單
            async submitOrder() {
                const submitBtn = document.getElementById('submitBtn');
                
                // 收集表單資料
                const formData = {
                    customer_name: document.getElementById('formName').value.trim(),
                    ig_account: document.getElementById('formIG').value.trim(),
                    phone: document.getElementById('formPhone').value.trim(),
                    email: document.getElementById('formEmail').value.trim(),
                    cvs_store_name: document.getElementById('formStore').value.trim(),
                };

                // 簡單驗證
                if (!formData.customer_name || !formData.ig_account || !formData.phone || !formData.cvs_store_name) {
                    this.showToast("請填寫所有必填欄位");
                    return;
                }

                // 準備訂單商品資料
                const orderItems = [];
                let subtotal = 0;
                this.data.products.forEach(p => {
                    const qty = this.data.cart[p.id] || 0;
                    if (qty > 0) {
                        const price = parseInt(p.price.replace(/\D/g, '')) || 0;
                        subtotal += price * qty;
                        orderItems.push({
                            id: p.id,
                            name: p.name,
                            price: price,
                            quantity: qty,
                            total: price * qty
                        });
                    }
                });

                const shipping = parseInt(this.data.settings.shipping_fee) || 60;
                
                const payload = {
                    action: 'submitOrder',
                    ...formData,
                    items_json: JSON.stringify(orderItems),
                    subtotal: subtotal,
                    shipping_fee: shipping,
                    total_price: subtotal + shipping
                };

                // UI 狀態：載入中
                submitBtn.disabled = true;
                submitBtn.innerHTML = '<div class="loader w-5 h-5 border-2 border-white border-top-transparent"></div> 處理中...';

                try {
                    if (!GAS_API_URL) {
                        // 模擬 API 延遲與成功
                        await new Promise(r => setTimeout(r, 1500));
                        this.finishOrder("TEST-" + Math.floor(Math.random()*10000));
                        return;
                    }

                    // 使用 fetch (POST) 送到 GAS。
                    // 注意：為了避免 CORS preflight，GAS 建議使用 text/plain 或 POST x-www-form-urlencoded
                    const response = await fetch(GAS_API_URL, {
                        method: 'POST',
                        body: JSON.stringify(payload)
                        // 不設定 Content-Type application/json 可繞過部分瀏覽器 CORS 嚴格檢查，GAS 依然能收到 e.postData.contents
                    });
                    
                    const result = await response.json();
                    
                    if (result.status === 'success') {
                        this.finishOrder(result.order_id);
                    } else {
                        throw new Error(result.message || "訂單送出失敗");
                    }
                } catch (error) {
                    console.error("提交失敗:", error);
                    this.showToast("發生錯誤，請重試或私訊 IG 聯繫我們。");
                    submitBtn.disabled = false;
                    submitBtn.innerHTML = '確認送出訂單';
                }
            },

            // 訂單完成處理
            finishOrder(orderId) {
                document.getElementById('successOrderId').innerText = orderId;
                // 清空購物車
                this.data.cart = {};
                this.updateCartUI();
                this.navigate('success');
            },

            // 共用工具：吐司提示
            showToast(msg) {
                const toast = document.getElementById('toast');
                toast.innerText = msg;
                toast.classList.remove('opacity-0');
                setTimeout(() => {
                    toast.classList.add('opacity-0');
                }, 3000);
            },

                const textArea = document.createElement("textarea");
